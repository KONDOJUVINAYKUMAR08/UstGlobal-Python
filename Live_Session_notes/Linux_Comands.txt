13.   read width and breadth of a rectangle.
      print area=w*b
      circum=2*(w+b)


	nano ust-gloab-6.sh
	read -p "Enter width and breadth of rectangle: "w b
	area = $((w * b)
	circum = $((2 * (w + b)))
	echo "Area of rectangle is: $area"
	echo "Circumference of rectangle is: $circum"

17. Sum of N Numbers n=10, sum=55  


	nano ust-gloab-10.sh

	read -p "Enter n: "n	
	i=1
	sum = 0
	while [$i -le $n]
	do
		sum=$((sum + i))
		i=$((i + 1))
	done
	echo "Sum = $sum"
